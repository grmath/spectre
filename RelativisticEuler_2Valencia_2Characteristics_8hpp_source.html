32 /*!
33  * \brief Compute the characteristic speeds for the Valencia formulation
34  * of the relativistic Euler system.
35  *
36  * The principal symbol of the system is diagonalized so that the elements of
37  * the diagonal matrix are the \f$(\text{Dim} + 2)\f$ characteristic speeds
38  *
39  * \f{align*}
40  * \lambda_1 &= \alpha \Lambda^- - \beta_n,\\
41  * \lambda_{i + 1} &= \alpha v_n - \beta_n,\quad i = 1,...,\text{Dim}\\
42  * \lambda_{\text{Dim} + 2} &= \alpha \Lambda^+ - \beta_n,
43  * \f}
44  *
45  * where \f$\alpha\f$ is the lapse, \f$\beta_n = n_i \beta^i\f$ and
46  * \f$v_n = n_i v^i\f$ are the projections of the shift \f$\beta^i\f$ and the
47  * spatial velocity \f$v^i\f$ onto the normal one-form \f$n_i\f$, respectively,
48  * and
49  *
50  * \f{align*}
51  * \Lambda^{\pm} &= \dfrac{1}{1 - v^2 c_s^2}\left[ v_n (1- c_s^2) \pm
52  * c_s\sqrt{\left(1 - v^2\right)\left[1 - v^2 c_s^2 - v_n^2(1 - c_s^2)\right]}
53  * \right],
54  * \f}
55  *
56  * where \f$v^2 = \gamma_{ij}v^iv^j\f$ is the magnitude squared of the spatial
57  * velocity, and \f$c_s\f$ is the sound speed.
58  */
59 template <size_t Dim>
60 void characteristic_speeds(
61  gsl::not_null<std::array<DataVector, Dim + 2>*> char_speeds,
62  const Scalar<DataVector>& lapse, const tnsr::I<DataVector, Dim>& shift,
63  const tnsr::I<DataVector, Dim>& spatial_velocity,
64  const Scalar<DataVector>& spatial_velocity_squared,
65  const Scalar<DataVector>& sound_speed_squared,
66  const tnsr::i<DataVector, Dim>& normal) noexcept;
67
